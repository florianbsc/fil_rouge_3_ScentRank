services:

  api:
    build:
      context: ./backend              # On construit l’image à partir du code backend
      target: prod                    # On utilise la target "prod" du Dockerfile (build optimisé, sans dev deps)
      dockerfile: ../Dockerfile       # Dockerfile multi-stage situé à la racine
    # container_name: nest_api_prod   # (Optionnel) Nom fixe pour le conteneur en prod
    ports:
      - "${PORT}:3000"                # Expose le port défini dans .env.prod sur le port 3000 du conteneur
    env_file: ../env/.env.prod               # Charge les variables d’environnement de prod
    depends_on:
      mongo:
        condition: service_healthy    # L’API démarre seulement quand MongoDB est prêt
    profiles: ["prod"]                # Active ce service uniquement si le profil "prod" est utilisé

  mongo:
    image: mongo:6                    # Image officielle MongoDB v6 (stable)
    # container_name: mongo_prod      # (Optionnel) Nom fixe pour le conteneur Mongo en prod
    restart: unless-stopped           # Le conteneur redémarre automatiquement sauf arrêt manuel
    env_file: .env.prod               # Variables d’env pour configurer Mongo (user, mdp, db…)
    ports:
      - "${MONGO_PORT}:27017"         # Expose MongoDB sur le port défini dans .env.prod
    volumes:
      - mongo_data_prod:/data/db      # Volume persistant pour stocker les données Mongo
    healthcheck:                      # Vérifie que Mongo est opérationnel avant de lancer l’API
      test: [ "CMD-SHELL", "mongosh --eval 'db.adminCommand({ ping: 1 })' || exit 1" ]
      interval: 5s                    # Vérifie toutes les 5 secondes
      timeout: 3s                     # Timeout max 3s pour la commande
      retries: 20                     # Retente jusqu’à 20 fois avant de déclarer Mongo KO

volumes:
  mongo_data_prod:                    # Volume pour persister la base Mongo en prod
